// Advanced Facebook Pixel Implementation
!function(f,b,e,v,n,t,s) {
  if(f.fbq) return;
  n=f.fbq=function(){
    n.callMethod ? n.callMethod.apply(n,arguments) : n.queue.push(arguments)
  };
  if(!f._fbq) f._fbq=n;
  n.push=n; n.loaded=!0; n.version='2.0';
  n.queue=[]; t=b.createElement(e); t.async=!0;
  t.src=v; s=b.getElementsByTagName(e)[0];
  s.parentNode.insertBefore(t,s)
}(window, document,'script','https://connect.facebook.net/en_US/fbevents.js');

function trackCustomEvent(eventName, parameters = {}) {
  fbq('track', eventName, parameters);
  console.log(`Tracked: ${eventName}`, parameters);
}

// Lead generation tracking
function trackLead(leadValue = 0) {
  trackCustomEvent('Lead', {

  });

  });
}

Google Ads Conversion Tracking

Enhanced Google Ads conversion tracking with dynamic values

google-ads.js 
// Google Ads Enhanced Conversion Tracking
class GoogleAdsTracker {
  constructor(conversionId, conversionLabel) {
    this.conversionId = conversionId;
    this.conversionLabel = conversionLabel;
    this.initGtag();
  }

  initGtag() {
    // Load gtag if not already loaded
    if (!window.gtag) {
      const script = document.createElement('script');
      script.async = true;
      script.src = `https://www.googletagmanager.com/gtag/js?id=${this.conversionId}`;
      document.head.appendChild(script);

      window.dataLayer = window.dataLayer || [];
      window.gtag = function() { dataLayer.push(arguments); };
      gtag('js', new Date());
      gtag('config', this.conversionId);
    }
  }

  trackConversion(value = 0, currency = 'USD', transactionId = null) {
    const conversionData = {
      'send_to': `${this.conversionId}/${this.conversionLabel}`,
      'value': value,
      'currency': currency
    };

    if (transactionId) {
      conversionData.transaction_id = transactionId;
    }

    gtag('event', 'conversion', conversionData);
    
    console.log('Google Ads Conversion Tracked:', conversionData);
  }

  trackEnhancedConversion(userEmail, phoneNumber = null) {
    gtag('event', 'conversion', {
      'send_to': `${this.conversionId}/${this.conversionLabel}`,
      'user_data': {
        'email_address': userEmail,
        'phone_number': phoneNumber
      }
    });
  }
}

// Usage example
const adsTracker = new GoogleAdsTracker('AW-XXXXXXXXX', 'conversion_label');
adsTracker.trackConversion(99.99, 'USD', 'TXN_' + Date.now());

Smart Lead Generation Form

Advanced form with validation, tracking, and CRM integration

lead-form.js 
class SmartLeadForm {
  constructor(formId, options = {}) {
    this.form = document.getElementById(formId);
    this.options = {
      validateEmail: true,
      trackEvents: true,
      showProgress: true,
      ...options
    };
    this.init();
  }

  init() {
    if (!this.form) return;
    
    this.addValidation();
    this.bindEvents();
  }

  bindEvents() {
    this.form.addEventListener('submit', (e) => {
      e.preventDefault();
      this.handleSubmit();
    });
  }

  addValidation() {
    const inputs = this.form.querySelectorAll('input, textarea, select');
    inputs.forEach(input => {
      input.addEventListener('blur', () => this.validateField(input));
      input.addEventListener('input', () => this.clearError(input));
    });
  }

  handleSubmit() {
    const formData = new FormData(this.form);
    const data = Object.fromEntries(formData.entries());
    
    // Validate all fields
    const inputs = this.form.querySelectorAll('input[required], textarea[required]');
    let isValid = true;
    
    inputs.forEach(input => {
      if (!this.validateField(input)) {
        isValid = false;
      }
    });
    
    if (isValid) {
      this.showSuccess('Form submitted successfully! (Demo mode)');
    }
  }

  validateField(field) {
    const value = field.value.trim();
    let isValid = true;
    let errorMessage = '';

    // Email validation
    if (field.type === 'email' && this.options.validateEmail) {
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!emailRegex.test(value)) {
        isValid = false;
        errorMessage = 'Please enter a valid email address';
      }
    }

    // Required field validation
    if (field.hasAttribute('required') && !value) {
      isValid = false;
      errorMessage = 'This field is required';
    }

    this.showFieldError(field, isValid ? '' : errorMessage);
    return isValid;
  }

  showFieldError(field, message) {
    let errorDiv = field.parentNode.querySelector('.error-message');
    
    if (!errorDiv) {
      errorDiv = document.createElement('div');
      errorDiv.className = 'error-message text-red-500 text-sm mt-1';
      field.parentNode.appendChild(errorDiv);
    }
    
    errorDiv.textContent = message;
    field.classList.toggle('border-red-500', !!message);
  }

  clearError(field) {
    const errorDiv = field.parentNode.querySelector('.error-message');
    if (errorDiv) errorDiv.textContent = '';
    field.classList.remove('border-red-500');
  }

  showSuccess(message) {
    const successDiv = document.createElement('div');
    successDiv.className = 'bg-green-100 border border-green-400 text-green-700 px-4 py-3 rounded mt-4';
    successDiv.textContent = message;
    
    // Remove any existing success/error messages
    const existing = this.form.parentNode.querySelector('.bg-green-100, .bg-red-100');
    if (existing) existing.remove();
    
    this.form.parentNode.appendChild(successDiv);
    
    // Auto-remove after 5 seconds
    setTimeout(() => {
      if (successDiv.parentNode) {
        successDiv.remove();
      }
    }, 5000);
  }

  showError(message) {
    const errorDiv = document.createElement('div');
    errorDiv.className = 'bg-red-100 border border-red-400 text-red-700 px-4 py-3 rounded mt-4';
    errorDiv.textContent = message;
    
    // Remove any existing success/error messages
    const existing = this.form.parentNode.querySelector('.bg-green-100, .bg-red-100');
    if (existing) existing.remove();
    
    this.form.parentNode.appendChild(errorDiv);
  }

  async submitForm(formData) {
    try {
      // Track form submission
      if (this.options.trackEvents) {
        if (typeof fbq !== 'undefined') fbq('track', 'Lead');
        if (typeof gtag !== 'undefined') gtag('event', 'generate_lead');
      }

      // Submit to your CRM/backend
      const response = await fetch('/api/leads', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(formData)
      });

      if (response.ok) {
        this.showSuccess('Thank you! We\'ll be in touch soon.');
      } else {
        throw new Error('Submission failed');
      }
    } catch (error) {
      this.showError('Something went wrong. Please try again.');
    }
  }
}

Performance Analytics Dashboard

Real-time campaign performance tracking with Chart.js visualization

analytics-dashboard.js 
class AnalyticsDashboard {
  constructor(containerId) {
    this.container = document.getElementById(containerId);
    this.charts = {};
    this.data = {
      campaigns: [],
      metrics: {},
      timeRange: '7d'
    };
    this.init();
  }

  async init() {
    await this.loadChartJS();
    this.createDashboard();
    this.fetchData();
    this.startRealTimeUpdates();
  }

  async loadChartJS() {
    if (!window.Chart) {
      const script = document.createElement('script');
      script.src = 'https://cdn.jsdelivr.net/npm/chart.js';
      document.head.appendChild(script);
      
      return new Promise(resolve => {
        script.onload = resolve;
      });
    }
  }

  createDashboard() {
    this.container.innerHTML = `
      

        

          
Total Leads

          
0

          +0%
        

        

          
ROAS

          
0.00x

          +0%
        

        

          
Cost Per Lead

          
$0

          +0%
        

        

          
Conversion Rate

          
0%

          +0%
        

      

      

        

          
Campaign Performance

          
        

        

          
Traffic Sources

          
        

      

    `;

    this.createCharts();
  }

  createCharts() {
    // Performance Chart
    const performanceCtx = document.getElementById('performance-chart');
    this.charts.performance = new Chart(performanceCtx, {
      type: 'line',
      data: {
        labels: [],
        datasets: [{
          label: 'Leads',
          data: [],
          borderColor: '#3b82f6',
          backgroundColor: 'rgba(59, 130, 246, 0.1)',
          tension: 0.4
        }]
      },
      options: {
        responsive: true,
        plugins: {
          legend: { display: false }
        },
        scales: {
          y: { beginAtZero: true }
        }
      }
    });

    // Sources Chart
    const sourcesCtx = document.getElementById('sources-chart');
    this.charts.sources = new Chart(sourcesCtx, {
      type: 'doughnut',
      data: {
        labels: ['Facebook Ads', 'Google Ads', 'Organic', 'Direct'],
        datasets: [{
          data: [0, 0, 0, 0],
          backgroundColor: ['#1877f2', '#4285f4', '#34a853', '#ea4335']
        }]
      },
      options: {
        responsive: true,
        plugins: {
          legend: { position: 'bottom' }
        }
      }
    });
  }

  async fetchData() {
    try {
      // Simulate API call
      const response = await fetch(`/api/analytics?range=${this.data.timeRange}`);
      const data = await response.json();
      
      this.updateMetrics(data.metrics);
      this.updateCharts(data.chartData);
    } catch (error) {
      console.error('Failed to fetch analytics data:', error);
    }
  }

  updateMetrics(metrics) {
    document.getElementById('total-leads').textContent = metrics.totalLeads || 0;
    document.getElementById('roas').textContent = `${metrics.roas || 0}x`;
    document.getElementById('cpl').textContent = `$${metrics.cpl || 0}`;
    document.getElementById('cvr').textContent = `${metrics.cvr || 0}%`;
  }

  startRealTimeUpdates() {
    setInterval(() => {
      this.fetchData();
    }, 30000); // Update every 30 seconds
  }
}

// Initialize dashboard
const dashboard = new AnalyticsDashboard('analytics-container');
